INPUTS		RKC, !RKC, !RK!C, RK!C, Reset;
OUTPUTS		TRUE, FALSE;
INITIAL_STATE	NS;
OTHER_STATES	SNRK, SRKBC, SRKGC;
TRANSITIONS	NS	: NS WHEN (!RK!C,FALSE) COST 1
			: SNRK WHEN (!RKC,TRUE) COST 1,
		SNRK	: SRKGC WHEN (RKC,TRUE) COST 1
			: NS WHEN (!RK!C,FALSE) COST 1,
		SRKGC	: SRKGC WHEN (RKC,TRUE) COST 1
			: SRKBC WHEN (RK!C,FALSE) COST 1
			: NS WHEN (Reset,TRUE) COST 1,
		SRKBC	: SRKBC WHEN (RK!C,FALSE) COST 1
			: SRKGC WHEN (RKC,TRUE) COST 1
			: NS WHEN (!RK!C,FALSE) COST 1
